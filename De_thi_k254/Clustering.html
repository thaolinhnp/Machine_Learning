In [20]:
from sklearn import metrics
list_sil = []
K = range(2,8)
for k in K:
    gmm = GaussianMixture(n_components=k)
    gmm.fit(data)
    labels = gmm.predict(data)
    sil = metrics.silhouette_score(data, labels, metric='euclidean')
    list_sil.append(sil)
