from sklearn import metrics
list_sil = []
K = range(2,8)
for k in K:
    gmm = GaussianMixture(n_components=k)
    gmm.fit(X_train)
    labels = gmm.predict(X_train)
    sil = metrics.silhouette_score(X_train, labels, metric='euclidean')
    list_sil.append(sil)
